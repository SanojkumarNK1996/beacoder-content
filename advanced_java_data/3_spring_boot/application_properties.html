⚙️ Application Properties

🎯 What are Application Properties?

Application Properties are external configuration files that allow you to customize your Spring Boot application behavior without changing code. They follow the "externalize configuration" principle for different environments.

🏠 Home Settings Analogy

Think of application properties like home settings:

  • 🌡️ Thermostat: Database connection settings
  • 🔊 Volume Control: Logging levels
  • 🚪 Security System: Authentication settings
  • 💡 Light Switches: Feature toggles

Change settings without rewiring the house!

📁 Configuration File Formats

📄 application.properties

Key-Value format - Traditional Java properties

# Server configuration
server.port=8081
server.servlet.context-path=/api

# Database configuration  
spring.datasource.url=jdbc:mysql://localhost:3306/mydb
spring.datasource.username=user
spring.datasource.password=secret

# JPA configuration
spring.jpa.hibernate.ddl-auto=update
spring.jpa.show-sql=true
spring.jpa.properties.hibernate.dialect=org.hibernate.dialect.MySQL8Dialect

# Logging
logging.level.com.example=DEBUG
logging.level.org.springframework.web=INFO
logging.file.name=application.log

📄 application.yml (Recommended)

YAML format - More readable and concise

# Server configuration
server:
  port: 8081
  servlet:
    context-path: /api

# Database configuration
spring:
  datasource:
    url: jdbc:mysql://localhost:3306/mydb
    username: user
    password: secret
  
  # JPA configuration
  jpa:
    hibernate:
      ddl-auto: update
    show-sql: true
    properties:
      hibernate:
        dialect: org.hibernate.dialect.MySQL8Dialect

# Logging
logging:
  level:
    com.example: DEBUG
    org.springframework.web: INFO
  file:
    name: application.log

🔧 Activating Profiles

1️⃣ In application.yml

# Set default active profile
spring:
  profiles:
    active: dev

2️⃣ Command Line Arguments

# Run with specific profile
java -jar myapp.jar --spring.profiles.active=prod

# Multiple profiles
java -jar myapp.jar --spring.profiles.active=prod,mysql

3️⃣ Environment Variables

# Set environment variable
export SPRING_PROFILES_ACTIVE=prod

# Run application
java -jar myapp.jar

# Docker environment
docker run -e SPRING_PROFILES_ACTIVE=prod myapp:latest

4️⃣ JVM System Properties

# JVM system property
java -Dspring.profiles.active=prod -jar myapp.jar

# IDE configuration
-Dspring.profiles.active=dev

💉 Property Injection in Code

1️⃣ @Value Annotation

@Service
public class EmailService {
    
    // Simple property injection
    @Value("${app.email.from}")
    private String fromEmail;
    
    // With default value
    @Value("${app.email.enabled:true}")
    private boolean emailEnabled;
    
    // System properties
    @Value("${java.home}")
    private String javaHome;
    
    // Expression evaluation
    @Value("#{systemProperties['user.name']}")
    private String userName;
    
    // List injection
    @Value("${app.features}")
    private List<String> features;
    
    public void sendEmail(String to, String message) {
        if (emailEnabled) {
            System.out.println("Sending email from: " + fromEmail);
            System.out.println("To: " + to);
            System.out.println("Message: " + message);
        }
    }
}
        
# application.yml for above service
app:
  email:
    from: noreply@mycompany.com
    enabled: true
  features:
    - user-registration
    - email-notifications
    - payment-processing

2️⃣ @ConfigurationProperties (Recommended)

// Type-safe configuration properties class
@ConfigurationProperties(prefix = "app.email")
@Data  // Lombok for getters/setters
@Component
public class EmailProperties {
    
    private String from = "noreply@example.com";
    private boolean enabled = true;
    private int timeout = 5000;
    private List<String> allowedDomains = new ArrayList<>();
    private Smtp smtp = new Smtp();
    
    @Data
    public static class Smtp {
        private String host = "localhost";
        private int port = 587;
        private boolean auth = true;
        private boolean starttls = true;
    }
}

// Usage in service
@Service
public class EmailService {
    
    private final EmailProperties emailProperties;
    
    public EmailService(EmailProperties emailProperties) {
        this.emailProperties = emailProperties;
    }
    
    public void sendEmail(String to, String message) {
        if (emailProperties.isEnabled()) {
            System.out.println("SMTP Host: " + emailProperties.getSmtp().getHost());
            System.out.println("SMTP Port: " + emailProperties.getSmtp().getPort());
            System.out.println("From: " + emailProperties.getFrom());
            // Send email logic...
        }
    }
}
        
# application.yml for type-safe properties
app:
  email:
    from: support@mycompany.com
    enabled: true
    timeout: 10000
    allowed-domains:
      - mycompany.com
      - partner.com
    smtp:
      host: smtp.gmail.com
      port: 587
      auth: true
      starttls: true

    password: password

    properties:
      hibernate:
        format_sql: true

🎯 Property Validation

Using @Validated and Bean Validation

@ConfigurationProperties(prefix = "app")
@Validated
@Data
@Component
public class AppProperties {
    
    @NotBlank(message = "Application name is required")
    private String name;
    
    @Min(value = 1, message = "Port must be greater than 0")
    @Max(value = 65535, message = "Port must be less than 65536")
    private int port = 8080;
    
    @Email(message = "Admin email must be valid")
    private String adminEmail;
    
    @Valid
    private Database database = new Database();
    
    @Data
    public static class Database {
        @NotBlank
        @Pattern(regexp = "^jdbc:.*", message = "Must be a valid JDBC URL")
        private String url;
        
        @NotBlank
        private String username;
        
        @Size(min = 8, message = "Password must be at least 8 characters")
        private String password;
    }
}
        
# application.yml with validation
app:
  name: "My Awesome App"              # Required
  port: 8080                          # Must be 1-65535
  admin-email: admin@example.com      # Must be valid email
  database:
    url: jdbc:mysql://localhost:3306/mydb  # Must start with jdbc:
    username: user                    # Required
    password: secretpassword123       # Min 8 characters

🔧 Advanced Configuration Techniques

1️⃣ Conditional Properties

# Enable features based on conditions
spring:
  profiles:
    active: dev
  
---
# Development profile
spring:
  config:
    activate:
      on-profile: dev
app:
  features:
    debug-mode: true
    fake-payment: true
    email-enabled: false

---
# Production profile  
spring:
  config:
    activate:
      on-profile: prod
app:
  features:
    debug-mode: false
    fake-payment: false
    email-enabled: true

2️⃣ Custom Property Sources

@Configuration
@PropertySource("classpath:custom.properties")
@PropertySource(value = "classpath:optional.properties", ignoreResourceNotFound = true)
public class CustomPropertiesConfig {
    
    @Bean
    public static PropertySourcesPlaceholderConfigurer propertyPlaceholder() {
        return new PropertySourcesPlaceholderConfigurer();
    }
}

📊 Property Precedence Order

Spring Boot resolves properties in this order (highest to lowest priority):

  1. 🔧 Command line arguments
  2. 🌍 OS environment variables
  3. ☕ Java System properties
  4. 📁 application-{profile}.properties outside jar
  5. 📁 application-{profile}.properties inside jar
  6. 📄 application.properties outside jar
  7. 📄 application.properties inside jar
  8. ⚙️ @PropertySource annotations
  9. 🔧 Default properties

🎯 Key Takeaway

Application Properties are the control center of your Spring Boot application. They enable environment-specific configuration without code changes, following the 12-Factor App methodology.

💡 Best Practice: Use YAML format for readability, @ConfigurationProperties for type safety, and environment variables for secrets. Always externalize configuration for different environments.
