🧭 Navigation and Programmatic Routing

🔗 Link and NavLink

import { Link, NavLink } from 'react-router-dom';

// Link: basic navigation
<Link to="/about">About</Link>
<Link to={`/users/${user.id}`}>View Profile</Link>

// NavLink: adds "active" CSS class when route matches
<NavLink to="/home" className={({ isActive }) => isActive ? 'active-link' : ''}>
  Home
</NavLink>

🚗 Programmatic Navigation with useNavigate

import { useNavigate } from 'react-router-dom';

function LoginForm() {
  const navigate = useNavigate();

  const handleLogin = async () => {
    await loginUser(credentials);
    navigate('/dashboard');           // Redirect after login
    navigate('/dashboard', { replace: true });  // Replace history (no back button)
    navigate(-1);                      // Go back
    navigate(1);                       // Go forward
  };
  ...
}

🔍 Query Parameters with useSearchParams

const [searchParams, setSearchParams] = useSearchParams();
const query = searchParams.get('q');       // /search?q=react
setSearchParams({ q: 'react hooks' });    // Update URL query

🎯 Summary

Use <Link> for navigation in JSX. Use useNavigate() for programmatic routing after events (form submit, login success).
