🔢 Route Parameters and Nested Routes

🔑 Route Parameters

// Define route with parameter
<Route path="/users/:userId" element={<UserProfile />} />
<Route path="/posts/:postId/comments/:commentId" element={<Comment />} />

// Access parameters in component
import { useParams } from 'react-router-dom';

function UserProfile() {
  const { userId } = useParams();
  // userId is the value from the URL: /users/42 → userId = '42'
  return <h1>User ID: {userId}</h1>;
}

🌳 Nested Routes

// Nested routes — shared layout
<Route path="/dashboard" element={<DashboardLayout />}>
  <Route index element={<DashboardHome />} />      {/* /dashboard */}
  <Route path="settings" element={<Settings />} />  {/* /dashboard/settings*/}
  <Route path="profile" element={<Profile />} />    {/* /dashboard/profile */}
</Route>

// DashboardLayout.jsx — renders child routes here:
import { Outlet } from 'react-router-dom';
function DashboardLayout() {
  return (
    <div>
      <Sidebar />
      <main><Outlet /></main>  {/* Child route renders here */}
    </div>
  );
}

🎯 Summary

Route params via :param + useParams(). Nested routes with <Outlet /> for shared layouts (dashboard sidebar, app shell, etc.).
