📅 Activity Selection & Interval Problems

🎯 Activity Selection Problem

Given n activities with start and end times, find the maximum number of non-overlapping activities.

Greedy choice: Always pick the activity that ends earliest and doesn't conflict with the last selected.

int maxActivities(int[] start, int[] end) {
    // Sort by end time
    Integer[] indices = IntStream.range(0, start.length).boxed().toArray(Integer[]::new);
    Arrays.sort(indices, (a, b) -> end[a] - end[b]);
    int count = 1, lastEnd = end[indices[0]];
    for (int i = 1; i < indices.length; i++) {
        if (start[indices[i]] >= lastEnd) { count++; lastEnd = end[indices[i]]; }
    }
    return count;
}
